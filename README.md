# Memory functions (implemented in calculator.c):
 *   Shift+1 = MS  (memory store)
 *   Shift+2 = MR  (memory recall)
 *   Shift+3 = MC  (memory clear)
 *   Shift+4 = M+  (add to memory)
 *   Shift+5 = M-  (subtract from memory)
